let contents = app.vault.getAbstractFileByPath(dv.current().file.path);
app.vault.read(contents).then(text => {
  let lines = text.split("\n");

  // Match any line with 4+ dashes as horizontal rule
  let ruleIndexes = lines
    .map((line, i) => /^-{3,}\s*$/.test(line.trim()) ? i : -1)
    .filter(i => i !== -1);

  // Determine cutoff: either second rule or end of file
  let cutoff = ruleIndexes.length >= 2 ? ruleIndexes[1] : lines.length;

  // Slice up to cutoff and count bullets. Make it line.trim to count sub lines, too.
  let bullets = lines.slice(0, cutoff).filter(line => line.startsWith("- ") && !line.startsWith("- [")); // bullet points, not checklists

  let itemCount = bullets.length;
